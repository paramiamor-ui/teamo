    <script>
        // Crear estrellas
        function createStars() {
            const container = document.getElementById('starsContainer');
            for (let i = 0; i < 100; i++) {
                const star = document.createElement('div');
                star.className = 'star';
                star.style.left = Math.random() * 100 + '%';
                star.style.top = Math.random() * 100 + '%';
                star.style.animationDelay = Math.random() * 3 + 's';
                container.appendChild(star);
            }
        }
        createStars();

        // Abrir carta al hacer clic en el teléfono
        const phoneWrapper = document.getElementById('phoneWrapper');
        const notification = document.getElementById('notification');
        const letter = document.getElementById('letter');

        notification.addEventListener('click', function() {
            phoneWrapper.style.display = 'none';
            letter.classList.add('show');
            if (!isPlaying) {
                toggleMusic();
            }
        });

        // Popup del mensaje original
        const messageHighlight = document.getElementById('messageHighlight');
        const popup = document.createElement('div');
        popup.className = 'message-popup';
        popup.innerHTML = `
            <div class="message-bubble sent">hola, ¿cómo estás?</div>
            <div class="message-bubble">¡Hola! Bien, ¿y tú? 😊</div>
            <div style="font-size: 0.8em; color: #636e72; text-align: center; margin-top: 10px;">27 de septiembre - El primer mensaje</div>
        `;
        messageHighlight.style.position = 'relative';
        messageHighlight.appendChild(popup);

        messageHighlight.addEventListener('mouseenter', () => popup.classList.add('show'));
        messageHighlight.addEventListener('mouseleave', () => popup.classList.remove('show'));

        // Efecto de explosión de corazones
        function createHeartExplosion() {
            const hearts = ['❤️', '💖', '💕', '💗', '💓', '💝'];
            const colors = ['#e74c3c', '#ff6b6b', '#feca57', '#48dbfb', '#ff9ff3'];
            
            for (let i = 0; i < 30; i++) {
                setTimeout(() => {
                    const heart = document.createElement('div');
                    heart.className = 'floating-heart';
                    heart.textContent = hearts[Math.floor(Math.random() * hearts.length)];
                    heart.style.left = (50 + (Math.random() - 0.5) * 50) + '%';
                    heart.style.bottom = '20%';
                    heart.style.color = colors[Math.floor(Math.random() * colors.length)];
                    heart.style.fontSize = (Math.random() * 20 + 20) + 'px';
                    document.body.appendChild(heart);
                    
                    setTimeout(() => heart.remove(), 4000);
                }, i * 100);
            }

            // Mensaje especial
            const btn = document.querySelector('.heart-btn');
            btn.textContent = '¡Amor enviado! 💕';
            setTimeout(() => {
                btn.textContent = '❤️ Enviar amor infinito ❤️';
            }, 3000);
        }

        // Control de música
        const musicBtn = document.getElementById('musicBtn');
        const bgMusic = document.getElementById('bgMusic');
        let isPlaying = false;

        function toggleMusic() {
            if (isPlaying) {
                bgMusic.pause();
                musicBtn.classList.remove('playing');
                musicBtn.textContent = '🎵';
            } else {
                bgMusic.play().catch(e => console.log('Error:', e));
                musicBtn.classList.add('playing');
                musicBtn.textContent = '🎶';
            }
            isPlaying = !isPlaying;
        }

        musicBtn.addEventListener('click', toggleMusic);

        // Efectos al hacer clic en palabras destacadas
        document.querySelectorAll('.highlight-text').forEach(text => {
            text.addEventListener('click', function(e) {
                const sparkle = document.createElement('div');
                sparkle.style.position = 'fixed';
                sparkle.style.left = e.clientX + 'px';
                sparkle.style.top = e.clientY + 'px';
                sparkle.style.width = '10px';
                sparkle.style.height = '10px';
                sparkle.style.background = '#e74c3c';
                sparkle.style.borderRadius = '50%';
                sparkle.style.pointerEvents = 'none';
                sparkle.style.animation = 'explode 0.5s forwards';
                document.body.appendChild(sparkle);
                setTimeout(() => sparkle.remove(), 500);
            });
        });

        // Animación de explosión
        const style = document.createElement('style');
        style.textContent = `
            @keyframes explode {
                0% { transform: scale(1); opacity: 1; }
                100% { transform: scale(3); opacity: 0; }
            }
        `;
        document.head.appendChild(style);
    </script>
